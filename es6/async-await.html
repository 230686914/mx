    # Async await

    需求场景：小名去考试，先考语文，在考数学，最后考英语。3科都考完之后，都及格给奖励。

    // 第一步考语文
var promise = new Promise((resolve, reject) => {
  // 使用定时器模拟
  setTimeout(() => {
    var res = 60;
    if(res >= 60){
      resolve();
    }else {
      reject();
    }
  })
})
// 第二步考数学
var promise1 = new Promise((resolve, reject) => {
  // 使用定时器模拟
  setTimeout(() => {
    var res = 70;
    if(res >= 60){
      resolve();
    }else {
      reject();
    }
  })
})
// 第三步考英语
var promise2 = new Promise((resolve, reject) => {
  // 使用定时器模拟
  setTimeout(() => {
    var res = 50;
    if(res >= 60){
      resolve();
    }else {
      reject();
    }
  })
})

// 考语文
promise.then(res => {
  console.log('语文考试及格')
  // 考数学
  promise1.then(res => {
    console.log('数学考试及格')
    // 考英语
    promise2.then(res => {
      console.log('英语考试及格')
      ...
         ...
            ...
               ...
    }).catch(e => {
      console.log('英语考试不及格')
    })
  }).catch(e => {
    console.log('数学考试不及格')
  })
}).catch(e => {
  console.log('语文考试不及格')
})

    上述代码写出来，代码很臃肿，会不断的回调嵌套

    基于上述情况，有人提出了 Async await

    // 使用 Async await 实现方式
var promise = new Promise((resolve, reject) => {
    // 使用定时器模拟
    setTimeout(() => {
      var res = 60;
      if(res >= 60){
        resolve('语文考试及格');
      }else {
        reject('语文考试不及格');
      }
    })
})
var promise1 = new Promise((resolve, reject) => {
  // 使用定时器模拟
  setTimeout(() => {
    var res = 70;
    if(res >= 60){
      resolve('数学考试及格');
    }else {
      reject('数学考试不及格');
    }
  })
})
var promise2 = new Promise((resolve, reject) => {
  // 使用定时器模拟
  setTimeout(() => {
    var res = 50;
    if(res >= 60){
      resolve('英语考试及格');
    }else {
      reject('英语考试不及格');
    }
  })
})
async function test() {
    // 先考语文，使用 await 来等待语文的结果
    // await 后面接收的是一个 Promise 对象
    // await promise - 会返回一个值，取决于 Promise 对象里面的 resolve, reject

    // 考语文
    var res = await promise; // 输出值为 语文考试及格
    // 考数学
    var res1 = await promise1; // 输出值为 数学考试及格
    // 考英语
    var res2 = await promise2; // 输出值为 英语考试不及格
    // ...
}
test()

    如果是多个异步嵌套，可以考虑用 Async await 来优化。
